# Supabase Configuration
# Copia este archivo a .env.local y rellena con tus credenciales
# Obtén estos valores desde tu proyecto en https://supabase.com
# Ve a Settings > API para encontrar la URL y las keys

# URL de tu proyecto Supabase
SUPABASE_URL=https://your-project.supabase.co

# Service Role Key (para uso en servidor - mantener secreta)
# Esta key tiene permisos completos, solo usarla en el servidor
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# Nota: Para desarrollo local, crea un archivo .env.local con estos valores
# Vercel automáticamente usará las variables de entorno configuradas en el dashboard
